const ProgressTracker = ({ stageIndex, stages }) => {
  const percent = (stageIndex / (stages.length - 1)) * 100;
  
  return (
    <div className="mt-5 mb-1 w-full">
      <div className="relative w-full h-1.5 bg-slate-200 rounded-full shadow-inner">
        <div 
          className="absolute top-0 left-0 h-full bg-gradient-to-r from-blue-500 to-sky-400 rounded-full transition-all duration-500 shadow-sm"
          style={{ width: `${percent}%` }}
        />
        {stages.map((_, idx) => (
          <div 
            key={idx} 
            className={`absolute top-1/2 -translate-y-1/2 w-2.5 h-2.5 rounded-full border-2 border-white shadow-sm transition-colors duration-500 ${idx <= stageIndex ? 'bg-blue-500' : 'bg-slate-300'}`}
            style={{ left: `calc(${(idx / (stages.length - 1)) * 100}% - 5px)` }}
          />
        ))}
      </div>
      <div className="flex justify-between mt-3 px-1">
        {stages.map((stage, idx) => (
          <span 
            key={stage} 
            className={`text-[9px] font-bold uppercase tracking-wider ${idx <= stageIndex ? 'text-blue-600' : 'text-slate-400'}`}
            style={{ textAlign: idx === 0 ? 'left' : idx === stages.length - 1 ? 'right' : 'center', width: '33%' }}
          >
            {stage}
          </span>
        ))}
      </div>
    </div>
  );
};

const SalaryTrendChart = ({ data, labels }) => {
  const max = 40; const min = 20; const height = 100; const width = 300;
  const stepX = width / (data.length - 1);
  const points = data.map((val, i) => `${i * stepX},${height - ((val - min) / (max - min)) * height}`).join(' ');

  return (
    <div className="w-full">
      <svg viewBox={`0 -15 ${width} ${height + 25}`} className="w-full h-auto overflow-visible">
        <defs>
          <linearGradient id="lineGrad" x1="0" y1="0" x2="1" y2="0"><stop offset="0%" stopColor="#3b82f6" /><stop offset="100%" stopColor="#0ea5e9" /></linearGradient>
          <linearGradient id="areaGrad" x1="0" y1="0" x2="0" y2="1"><stop offset="0%" stopColor="#3b82f6" stopOpacity="0.2" /><stop offset="100%" stopColor="#0ea5e9" stopOpacity="0" /></linearGradient>
        </defs>
        <line x1="0" y1="0" x2={width} y2="0" stroke="#f1f5f9" strokeWidth="1.5" />
        <line x1="0" y1={height/2} x2={width} y2={height/2} stroke="#f1f5f9" strokeWidth="1.5" />
        <line x1="0" y1={height} x2={width} y2={height} stroke="#f1f5f9" strokeWidth="1.5" />
        <polygon points={`0,${height} ${points} ${width},${height}`} fill="url(#areaGrad)" />
        <polyline points={points} fill="none" stroke="url(#lineGrad)" strokeWidth="3" strokeLinecap="round" strokeLinejoin="round" />
        {data.map((val, i) => {
          const x = i * stepX; const y = height - ((val - min) / (max - min)) * height;
          return (
            <g key={i}>
              <circle cx={x} cy={y} r="4" fill="#fff" stroke="#0ea5e9" strokeWidth="2.5" />
              {i === data.length - 1 && <text x={x} y={y - 12} fill="#0284c7" fontSize="13" fontWeight="bold" textAnchor="end">₹{val}L</text>}
            </g>
          );
        })}
      </svg>
      <div className="flex justify-between mt-3 px-1 text-[10px] font-bold text-slate-400">{labels.map(lbl => <span key={lbl}>{lbl}</span>)}</div>
    </div>
  );
};

const SwipeableJobCard = ({ job, onSwipe, openDetails, isTop, index }) => {
  const [offset, setOffset] = useState({ x: 0, y: 0 });
  const [isDragging, setIsDragging] = useState(false);
  const cardRef = useRef(null);
  const startPos = useRef({ x: 0, y: 0 });

  const handlePointerDown = (e) => {
    if (!isTop) return;
    setIsDragging(true);
    const clientX = e.touches ? e.touches[0].clientX : e.clientX;
    const clientY = e.touches ? e.touches[0].clientY : e.clientY;
    startPos.current = { x: clientX - offset.x, y: clientY - offset.y };
  };

  const handlePointerMove = (e) => {
    if (!isDragging || !isTop) return;
    const clientX = e.touches ? e.touches[0].clientX : e.clientX;
    const clientY = e.touches ? e.touches[0].clientY : e.clientY;
    setOffset({ x: clientX - startPos.current.x, y: clientY - startPos.current.y });
  };

  const handlePointerUp = () => {
    if (!isDragging || !isTop) return;
    setIsDragging(false);
    
    if (Math.abs(offset.x) < 5 && Math.abs(offset.y) < 5) {
      openDetails(job);
      return;
    }

    const threshold = window.innerWidth * 0.25;
    if (offset.x > threshold) animateSwipe('right');
    else if (offset.x < -threshold) animateSwipe('left');
    else setOffset({ x: 0, y: 0 });
  };

  const animateSwipe = (direction) => {
    const endX = direction === 'right' ? window.innerWidth : -window.innerWidth;
    setOffset({ x: endX, y: offset.y });
    setTimeout(() => { onSwipe(direction, job); }, 300);
  };

  useEffect(() => {
    if (!isTop) return;
    const handleProgrammaticSwipe = (e) => {
      if (e.detail.id === job.id) animateSwipe(e.detail.direction);
    };
    window.addEventListener('triggerSwipe', handleProgrammaticSwipe);
    return () => window.removeEventListener('triggerSwipe', handleProgrammaticSwipe);
  }, [isTop, job.id, offset.y]);

  const rotation = offset.x * 0.05;
  const opacity = isTop ? 1 : 1 - (index * 0.15);
  const scale = isTop ? 1 : 0.95 - (index * 0.05);
  const translateY = isTop ? 0 : index * 12;

  const cardStyle = {
    transform: isTop ? `translate3d(${offset.x}px, ${offset.y}px, 0) rotate(${rotation}deg)` : `translate3d(0, ${translateY}px, 0) scale(${scale})`,
    transition: isDragging ? 'none' : 'transform 0.3s cubic-bezier(0.2, 0.8, 0.2, 1), opacity 0.3s',
    opacity, zIndex: 10 - index
  };

  const applyOpacity = Math.min(Math.max(offset.x / 100, 0), 1);
  const saveOpacity = Math.min(Math.max(-offset.x / 100, 0), 1);

  return (
    <div 
      ref={cardRef}
      className="absolute inset-0 w-full h-full touch-none prevent-select bg-white/95 backdrop-blur-3xl rounded-[2.5rem] border border-white shadow-[0_20px_40px_rgba(0,0,0,0.08)] flex flex-col overflow-hidden will-change-transform cursor-pointer"
      style={cardStyle}
      onMouseDown={handlePointerDown} onMouseMove={handlePointerMove} onMouseUp={handlePointerUp} onMouseLeave={handlePointerUp}
      onTouchStart={handlePointerDown} onTouchMove={handlePointerMove} onTouchEnd={handlePointerUp}
    >
      <div className="absolute inset-0 bg-blue-500/15 z-30 pointer-events-none transition-opacity" style={{ opacity: applyOpacity }}>
        <div className="absolute top-1/3 left-6 border-4 border-blue-500 text-blue-500 font-black text-3xl uppercase tracking-widest px-6 py-2 rounded-2xl rotate-[-15deg] bg-white/90 backdrop-blur-md shadow-2xl">Apply</div>
      </div>
      <div className="absolute inset-0 bg-slate-500/15 z-30 pointer-events-none transition-opacity" style={{ opacity: saveOpacity }}>
        <div className="absolute top-1/3 right-6 border-4 border-slate-500 text-slate-500 font-black text-3xl uppercase tracking-widest px-6 py-2 rounded-2xl rotate-[15deg] bg-white/90 backdrop-blur-md shadow-2xl">Save</div>
      </div>

      <div className="relative h-[45%] w-full shrink-0 overflow-hidden bg-slate-100">
        <img src={job.imageUrl} alt={job.company} className="w-full h-full object-cover" />
        <div className="absolute inset-0 bg-gradient-to-t from-white via-white/5 to-black/30" />
        <div className="absolute top-5 right-5"><MatchBadgeOverlay score={job.match} /></div>
      </div>

      <div className="px-6 pb-5 pt-0 flex-1 flex flex-col bg-white relative">
        <div className="absolute -top-10 left-6">
          <div className="w-16 h-16 bg-white rounded-2xl p-1.5 shadow-[0_4px_15px_rgba(0,0,0,0.08)] border-2 border-white/80">
            <img src={job.companyLogo} alt={job.company} className="w-full h-full object-contain rounded-xl" />
          </div>
        </div>

        <div className="mt-8 mb-3">
          <h2 className="text-[22px] font-black text-slate-800 leading-tight tracking-tight mb-1 line-clamp-2">{job.title}</h2>
          <div className="flex items-center gap-2">
            <span className="text-[15px] font-bold text-slate-500">{job.company}</span>
            <span className="w-1 h-1 bg-slate-300 rounded-full" />
            <span className="bg-sky-50 text-blue-600 px-2 py-0.5 rounded-md text-[10px] font-black uppercase tracking-wider border border-sky-100/50">{job.type}</span>
          </div>
        </div>

        <p className="text-[13px] text-slate-600 font-medium leading-relaxed line-clamp-3 mb-4">{job.description}</p>

        <div className="grid grid-cols-2 gap-3 mt-auto mb-2">
          <div className="bg-slate-50/80 border border-slate-100 p-3 rounded-2xl flex flex-col justify-center shadow-sm">
            <div className="flex items-center gap-1.5 mb-1 text-slate-500">
              <MapPin size={14} className="text-blue-500" />
              <span className="text-[10px] font-bold uppercase tracking-wider">Location</span>
            </div>
            <span className="text-[12px] font-bold text-slate-800 truncate">{job.location}{job.area ? `, ${job.area}` : ''}</span>
          </div>
          <div className="bg-slate-50/80 border border-slate-100 p-3 rounded-2xl flex flex-col justify-center shadow-sm">
            <div className="flex items-center gap-1.5 mb-1 text-slate-500">
              <div className="w-3.5 h-3.5 rounded-full bg-emerald-100 text-emerald-600 flex items-center justify-center text-[10px] font-black">₹</div>
              <span className="text-[10px] font-bold uppercase tracking-wider">Package</span>
            </div>
            <span className="text-[12px] font-bold text-slate-800 truncate">{job.salary}</span>
          </div>
        </div>
      </div>
    </div>
  );
};

          <button onClick={() => setTempSelected([])} className="px-6 py-4 rounded-2xl font-bold text-[15px] text-slate-500 hover:text-slate-700 hover:bg-slate-50 transition border border-transparent">Clear</button>

        </div>
      </div>
    </div>
  );
};

        {isSaved ? 'Saved' : 'Save'}
      </button>

const SearchScreen = ({ allJobs, savedJobs, appliedJobs, onSave, onApply, openDetails }) => {
  const [filters, setFilters] = useState({ jobType: [], location: [], salary: [], size: [] });
  const [searchQuery, setSearchQuery] = useState("");
  const [activeSheet, setActiveSheet] = useState(null);

  const handleApplyFilter = (key, selected) => { setFilters(prev => ({ ...prev, [key]: selected })); setActiveSheet(null); };
  const hasFilters = Object.values(filters).some(arr => arr.length > 0) || searchQuery.trim().length > 0;

  const filteredJobs = allJobs.filter(job => {
    let match = true;
    if (searchQuery) {
      const query = searchQuery.toLowerCase();
      if (!job.title.toLowerCase().includes(query) && !job.company.toLowerCase().includes(query)) match = false;
    }
    if (filters.jobType.length > 0 && !filters.jobType.includes(job.type)) match = false;
    if (filters.location.length > 0 && !filters.location.some(l => job.location.includes(l))) match = false;
    if (filters.size.length > 0 && !filters.size.includes(job.companySize)) match = false;
    return match;
  });

  return (
    <div className="flex-1 bg-gradient-to-br from-slate-50 to-sky-50 flex flex-col relative pb-28">
      <div className="bg-white/70 backdrop-blur-2xl border-b border-white/80 shadow-[0_4px_20px_rgba(0,0,0,0.02)] shrink-0 z-30 sticky top-0 pt-12">
        <div className="px-6 pb-2 text-center">
          <h1 className="text-[17px] font-black text-slate-800 tracking-wide">Search</h1>
          <p className="text-[11px] font-semibold text-slate-500 mt-0.5">Find roles that match your exact preferences</p>
        </div>
        <div className="px-6 py-3">
          <div className="flex items-center gap-3 bg-white/80 backdrop-blur-xl p-3.5 rounded-2xl border border-white shadow-[0_4px_15px_rgb(0,0,0,0.03)] focus-within:shadow-[0_4px_20px_rgba(59,130,246,0.1)] focus-within:border-blue-200 transition-all">
            <Search size={20} className="text-blue-500" />
            <input 
              type="text" placeholder="Search roles, companies..." value={searchQuery} onChange={(e) => setSearchQuery(e.target.value)}
              onKeyDown={(e) => { if (e.key === 'Enter') e.target.blur(); }}
              className="bg-transparent border-none outline-none w-full text-[15px] font-bold text-slate-800 placeholder:text-slate-400 placeholder:font-medium" 
            />
            {searchQuery && <button onClick={()=>setSearchQuery("")}><XIcon size={16} className="text-slate-400"/></button>}
          </div>
        </div>

        <div className="flex gap-2.5 overflow-x-auto hide-scrollbar px-6 pb-4">
          {Object.keys(FILTER_LABELS).map(key => {
            const count = filters[key].length;
            const isActive = count > 0;
            return (
              <button 
                key={key} onClick={() => setActiveSheet(key)}
                className={`shrink-0 flex items-center gap-1.5 px-4 py-2 rounded-xl text-[13px] font-bold border transition shadow-sm ${isActive ? 'bg-blue-50 border-blue-200 text-blue-700' : 'bg-white border-slate-200 text-slate-600 hover:bg-slate-50'}`}
              >
                {FILTER_LABELS[key]}
                {isActive && <span className="bg-blue-600 text-white w-4 h-4 rounded-full flex items-center justify-center text-[9px] ml-1">{count}</span>}
                <ChevronDown size={14} className={isActive ? 'text-blue-500' : 'text-slate-400'} />
              </button>
            );
          })}
        </div>
      </div>

      <div className="flex-1 overflow-y-auto hide-scrollbar p-6">
        <div className="flex justify-between items-end mb-6">
          <h2 className="text-[18px] font-black text-slate-800">{hasFilters ? 'Refined Results' : 'Explore All'}</h2>
          <span className="text-[12px] font-bold text-slate-400">{filteredJobs.length} jobs</span>
        </div>

        {filteredJobs.map(job => (
          <SearchResultCard 
            key={job.id} job={job} openDetails={openDetails} onSave={onSave} onApply={onApply} 
            isSaved={savedJobs.some(j => j.id === job.id)} isApplied={appliedJobs.some(j => j.id === job.id)}
          />
        ))}
        
        {filteredJobs.length === 0 && (
          <div className="flex flex-col items-center justify-center py-16 text-center animate-pop">
             <div className="w-20 h-20 bg-white/60 backdrop-blur-md rounded-full flex items-center justify-center text-blue-500 mb-6 shadow-[0_8px_30px_rgb(0,0,0,0.04)] border border-white">
               <Search size={32} strokeWidth={1.5} />
             </div>
             <h2 className="text-[20px] font-black text-slate-800 mb-2 tracking-tight">No matching roles found</h2>
             <p className="text-[14px] text-slate-500 font-medium leading-relaxed max-w-[250px]">Try changing your filters or searching broader terms.</p>
          </div>
        )}
      </div>

      {activeSheet && (
        <FilterBottomSheet filterKey={activeSheet} title={FILTER_LABELS[activeSheet]} options={FILTER_OPTIONS[activeSheet]} selected={filters[activeSheet]} onApply={handleApplyFilter} onClose={() => setActiveSheet(null)} />
      )}
    </div>
  );
};


// --- MAIN APP (State & Routing) ---

export default function App() {
  const [currentScreen, setCurrentScreen] = useState('splash');
  const [deck, setDeck] = useState(baseJobs);
  const [savedJobs, setSavedJobs] = useState([]);
  const [appliedJobs, setAppliedJobs] = useState([
    { ...baseJobs[0], id: 101, isActive: true, status: "Screening", appliedAt: "Applied 3 days ago", updatedAt: "Updated yesterday", stageIndex: 1, stages: ["Applied", "Screening", "Interview", "Offer"] },
    ...mockPastApplications
  ]); 
  const [chats, setChats] = useState(mockInitialChats);
  const [user, setUser] = useState(initialUser);
  const [documents, setDocuments] = useState([{ id: 1, name: "Arjun_Resume_v4.pdf", date: "2 days ago", type: "PDF" }]);
  const [settings, setSettings] = useState({ notifications: true, darkMode: false });
  
  const [selectedJobDetails, setSelectedJobDetails] = useState(null);
  const [selectedChat, setSelectedChat] = useState(null);

  const handleApply = (job) => {
    if (!appliedJobs.find(j => j.id === job.id)) {
      const newApp = { 
        ...job, isActive: true, status: "Applied", appliedAt: "Just now", updatedAt: "Just now", stageIndex: 0, stages: ["Applied", "Screening", "Interview", "Offer"] 
      };
      setAppliedJobs(prev => [newApp, ...prev]);
    }
    setDeck(prev => prev.filter(j => j.id !== job.id));
    setSavedJobs(prev => prev.filter(j => j.id !== job.id));
  };

  const handleSave = (job) => {
    if (!savedJobs.find(j => j.id === job.id)) setSavedJobs(prev => [job, ...prev]);
    setDeck(prev => prev.filter(j => j.id !== job.id));
  };

  const handleSwipe = (direction, job) => {
    direction === 'right' ? handleApply(job) : handleSave(job);
  };

  const handleSendMessage = (chatId, text) => {
    setChats(prevChats => prevChats.map(c => {
      if (c.id === chatId) {
        return { 
          ...c, lastMessage: text, time: "Just now", unreadCount: 0,
          messages: [...c.messages, { id: Date.now(), text, sender: 'user', time: 'Just now' }] 
        };
      }
      return c;
    }));
  };

  const handleOpenChat = (chat) => {
    setChats(prev => prev.map(c => c.id === chat.id ? { ...c, unreadCount: 0 } : c));
    setSelectedChat(chat);
  };

  const navigateTo = (screen) => {
    setCurrentScreen(screen);
    setSelectedChat(null);
    window.scrollTo(0, 0);
  };

  const totalUnread = chats.reduce((sum, chat) => sum + chat.unreadCount, 0);
  const showBottomNav = ['discover', 'search', 'applications', 'messages', 'profile', 'saved', 'insights'].includes(currentScreen);

  const BottomNav = () => (
    <div className="absolute bottom-0 w-full bg-white/70 backdrop-blur-2xl border-t border-white/80 px-6 py-4 flex justify-between items-center z-40 pb-8 shadow-[0_-10px_40px_rgba(0,0,0,0.03)]">
      {[
        { id: 'discover', icon: Home },
        { id: 'search', icon: Search },
        { id: 'applications', icon: Briefcase },
        { id: 'messages', icon: MessageSquare, badge: totalUnread },
        { id: 'profile', icon: User }
      ].map(item => {
        const Icon = item.icon;
        const isActive = currentScreen === item.id || (item.id === 'profile' && ['saved', 'insights', 'documents', 'settings', 'subscription'].includes(currentScreen));
        return (
          <button key={item.id} onClick={() => navigateTo(item.id)} className={`p-3 rounded-2xl transition-all duration-300 relative ${isActive ? 'bg-blue-50 text-blue-600 shadow-sm scale-110' : 'text-slate-400 hover:bg-slate-50 hover:text-slate-600'}`}>
            <Icon size={24} strokeWidth={isActive ? 2.5 : 2} />
            {item.badge > 0 && <span className="absolute top-1.5 right-1.5 w-3.5 h-3.5 bg-blue-500 text-white text-[8px] font-bold rounded-full border-2 border-white flex items-center justify-center">{item.badge}</span>}
          </button>
        )
      })}
    </div>
  );

  const renderScreen = () => {
    if (selectedChat) return <ChatDetailScreen chat={chats.find(c=>c.id === selectedChat.id)} onBack={() => setSelectedChat(null)} onSendMessage={handleSendMessage} />;
    
    switch(currentScreen) {
      case 'splash': return <SplashScreen nav={navigateTo} />;
      case 'welcome': return <WelcomeScreen nav={navigateTo} />;
      case 'onboarding_prefs': return <OnboardingPreferencesScreen nav={navigateTo} />;
      case 'onboarding_upload': return <OnboardingUploadScreen nav={navigateTo} />;
      case 'onboarding_review': return <OnboardingReviewScreen nav={navigateTo} />;
      case 'discover': return <SwipeDiscoveryScreen deck={deck} onSwipe={handleSwipe} openDetails={setSelectedJobDetails} />;
      case 'search': return <SearchScreen allJobs={baseJobs} savedJobs={savedJobs} appliedJobs={appliedJobs} onSave={handleSave} onApply={handleApply} openDetails={setSelectedJobDetails} />;
      case 'applications': return <ApplicationsScreen appliedJobs={appliedJobs} />;
      case 'messages': return <MessagesListScreen chats={chats} onChatSelect={handleOpenChat} />;
      case 'profile': return <ProfileScreen user={user} nav={navigateTo} />;
      case 'insights': return <MarketInsightsScreen user={user} onBack={() => navigateTo('profile')} />;
      case 'saved': return <SavedJobsScreen savedJobs={savedJobs} onApply={handleApply} onRemove={(job) => setSavedJobs(prev => prev.filter(j => j.id !== job.id))} openDetails={setSelectedJobDetails} onBack={() => navigateTo('profile')} nav={navigateTo} />;
      case 'documents': return <DocumentsScreen documents={documents} setDocuments={setDocuments} onBack={() => navigateTo('profile')} />;
      case 'settings': return <SettingsScreen settings={settings} setSettings={setSettings} onBack={() => navigateTo('profile')} onLogout={()=>navigateTo('welcome')} />;
      case 'subscription': return <SubscriptionScreen user={user} onUpgrade={() => setUser({...user, isPro: true})} onBack={() => navigateTo('profile')} />;
      default: return <SplashScreen nav={navigateTo} />;
    }
  };

  const isJobApplied = appliedJobs.some(j => j.id === selectedJobDetails?.id);
  const isJobSaved = savedJobs.some(j => j.id === selectedJobDetails?.id);

  return (
    <div className="min-h-screen bg-slate-100 flex items-center justify-center p-4 font-sans selection:bg-blue-100 overflow-x-hidden">
      <GlobalStyles />
      <div className="w-full max-w-[400px] h-[850px] bg-white rounded-[3rem] shadow-[0_20px_60px_rgba(0,0,0,0.1)] relative overflow-hidden flex flex-col border-[8px] border-slate-200 ring-4 ring-slate-100/50">
        <div className="absolute top-0 left-1/2 -translate-x-1/2 w-[120px] h-[30px] bg-slate-900 rounded-b-3xl z-50 flex justify-center items-center shadow-md">
          <div className="w-16 h-1.5 bg-slate-800 rounded-full"></div>
        </div>
        <div className="h-10 w-full bg-transparent z-40 absolute top-0 pointer-events-none flex justify-between px-6 items-end pb-1 text-[11px] font-bold shrink-0">
          <span className="text-slate-800 mix-blend-overlay">9:41</span>
        </div>

        {renderScreen()}

        {selectedJobDetails && <JobDetailsSheet job={selectedJobDetails} onClose={() => setSelectedJobDetails(null)} onApply={handleApply} onSave={handleSave} isApplied={isJobApplied} isSaved={isJobSaved} />}

        {showBottomNav && !selectedChat && <BottomNav />}
        
        <div className="absolute bottom-2 left-1/2 -translate-x-1/2 w-[120px] h-1.5 bg-slate-300 rounded-full z-50 pointer-events-none"></div>
      </div>
    </div>
  );
}
